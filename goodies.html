Simple tests

    def is_even(n):
        return bool(n % 2 == 0)

    def test_four_is_even():
        assert is_even(4)

    def test_five_is_not_even():
        assert not is_even(5)

Parameterization

    import pytest

    @pytest.mark.parametrize(
        "n, expected",
        [
            (4, True),
            (5, False),
            (0, True),
            (-3, False),
            (-2, True),
        ],
    )
    def test_is_even(n, expected):
        assert is_even(n) is expected
