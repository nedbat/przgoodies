Unified parameterization

Simple tests

A function:

    def is_even(n):
        return bool(n % 2 == 0)

Tests:

    def test_four_is_even():
        assert is_even(4)

    def test_five_is_not_even():
        assert not is_even(5)

Parameterization

    import pytest

    @pytest.mark.parametrize(
        "n, expected",
        [
            (4, True),
            (5, False),
            (0, True),
            (-3, False),
        ],
    )
    def test_is_even(n, expected):
        assert is_even(n) is expected

Combined!

    from contextlib import nullcontext as produces
    from pytest import raises

    @pytest.mark.parametrize(
        "x, result",
        [
            (4,     produces(1.5)),
            (3,     produces(2.0)),
            (2,     produces(3.0)),
            (0,     raises(ZeroDivisionError)),
            ("cat", raises(TypeError)),
        ],
    )
    def test_division(x, result):
        with result as expected:
            assert six_divided(x) == expected

Context managers

Usually:

    with open("myfile.txt") as f:
        data = f.read()

This is the same:

    opened_file = open("myfile.txt")
    with opened_file as f:
        data = f.read()

nullcontext

    from contextlib import nullcontext

    with nullcontext(42) as value:
        print(value)  # prints 42

Meant for conditional situations:

    from contextlib import nullcontext, redirect_stdout

    if to_file:
        out_ctrl = redirect_stdout(open("out.txt", "w"))
    else:
        out_ctrl = nullcontext()

    with out_ctrl:
        print("Hello world!")
